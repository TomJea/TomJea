require 'sqlite3'
DB = SQLite3::Database.new('doctors.db')
class Doctor
attr_reader :id, :age, :name, :specialty
def initialize(att={})
@id=att[:id] || nil
@name = att[:name]
@age = att[:age]
@specialty=att[:specialty] || "General consultation"
end
########################
#Instance classes
########################
def add
DB.execute("INSERT INTO doctors(name,age,specialty) VALUES('#{@name}',#{@age},'#{@specialty}')")
end
########################
#Method classes
########################
def self.general_inf
DB.results_as_hash = true
query = DB.execute("SELECT * FROM doctors")
query.each{|doctor| puts "#{doctor['id']} | The doctor #{doctor['name']} is #{doctor['age']} and his/her specialty is #{doctor['specialty']}"}
end
def self.remove(id)
DB.execute("DELETE FROM doctors WHERE id = #{id}")
end
end
